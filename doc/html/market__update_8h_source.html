83 uint64_t lastUpdateId = std::numeric_limits<uint64_t>::max();
84 void AddToQueue(EventLFQueue& queue) {
85 std::vector<MEMarketUpdateDouble> bulk;
86 bulk.resize(bids.size());
87 int i = 0;
88 for (auto& bid : bids) {
89 MEMarketUpdateDouble event;
90 event.ticker = ticker;
91 event.side = Common::Side::SELL;
92 event.price = bid.price;
93 event.qty = bid.qty;
94 bulk[i] = event;
95 i++;
96 }
97 bool status = false;
98 while(!status) status = queue.try_enqueue_bulk(&bulk[0], bids.size());
99
100 bulk.resize(asks.size());
101 i = 0;
102 for (auto& ask : asks) {
103 MEMarketUpdateDouble event;
104 event.ticker = ticker;
105 event.side = Common::Side::BUY;
106 event.price = ask.price;
107 event.qty = ask.qty;
108 bulk[i] = event;
109 i++;
110 }
111 status = false;
112 while(!status) status = queue.try_enqueue_bulk(&bulk[0], asks.size());
113 }
114 auto ToString() const {
115 return fmt::format("BookSnapshot[lastUpdateId:{}]", lastUpdateId);
116 };
