51 const auto side_index = Common::sideToIndex(client_response->side);
52 const auto opp_side_index = Common::sideToIndex(
53 client_response->side == Side::BUY ? Side::SELL : Side::BUY);
54 const auto side_value = Common::sideToValue(client_response->side);
55 position += client_response->exec_qty * side_value;
56 volume += client_response->exec_qty;
57
58 if (old_position * sideToValue(client_response->side) >=
59 0) { // opened / increased position.
60 open_vwap[side_index] +=
61 (client_response->price * client_response->exec_qty);
62 } else { // decreased position.
63 const auto opp_side_vwap =
64 open_vwap[opp_side_index] / std::abs(old_position);

75 open_vwap[opp_side_index] = 0;
