422class FactoryRequest {
423 public:
424 explicit FactoryRequest(const https::ExchangeI* exchange,
425 std::string_view end_point, const ArgsQuery& args,
426 boost::beast::http::verb action, SignerI* signer,
427 bool need_sign = false)
428 : exchange_(exchange), args_(args), action_(action), signer_(signer) {
429 if (need_sign) {
430 AddSignParams();
431 auto request_args = args_.QueryString();
432 auto signature = signer_->Sign(request_args);
433 AddSignature(signature);
434 }
435 end_point_ = end_point.data() + args_.FinalQueryString();
436 };
437 boost::beast::http::request<boost::beast::http::string_body> operator()() {
438 boost::beast::http::request<boost::beast::http::string_body> req;
439 req.version(11);
440 req.method(action_);
441 req.target(end_point_);
447 if (signer_) req.insert("X-MBX-APIKEY", signer_->ApiKey());
448 req.set(boost::beast::http::field::host, exchange_->Host().data());
449 req.set(boost::beast::http::field::user_agent,
450 BOOST_BEAST_VERSION_STRING);
458 req.set(boost::beast::http::field::content_type,
459 "application/x-www-form-urlencoded");
460 return req;
461 };
462 std::string_view Host() const { return exchange_->Host(); };
463 std::string_view Port() const { return exchange_->Port(); };
464 std::string_view EndPoint() const { return end_point_; }
465
466 private:
467 void AddSignParams() {
468 CurrentTime time_service;
469 args_["recvWindow"] = std::to_string(exchange_->RecvWindow());
470 args_["timestamp"] = std::to_string(time_service.Time());
471 };
472 void AddSignature(std::string_view signature) {
473 args_["signature"] = signature.data();
474 };
475
476 private:
477 const https::ExchangeI* exchange_;
478 std::string end_point_;
479 ArgsQuery args_;
480 boost::beast::http::verb action_;
481 SignerI* signer_;
482};
