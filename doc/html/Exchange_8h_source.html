83 static const char hex[] = "0123456789abcdef";
84 std::string res;
85 res.reserve(n * 2);
86
87 for (auto end = p + n; p != end; ++p) {
88 const std::uint8_t v = (*p);
89 res += hex[(v >> 4) & 0x0F];
90 res += hex[v & 0x0F];
91 }
